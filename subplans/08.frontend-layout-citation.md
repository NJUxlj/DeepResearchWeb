### `frontend/src/components/layout/ReferencePanel.tsx`

```tsx
import { X, ExternalLink, Globe, Database, Brain, FileText } from "lucide-react";
import { useReferenceStore } from "@/stores/referenceStore";
import type { Citation } from "@/types/session";

interface ReferencePanelProps {
  isOpen: boolean;
  onClose: () => void;
  className?: string;
}

export const ReferencePanel: React.FC<ReferencePanelProps> = ({
  isOpen,
  onClose,
  className,
}) => {
  const { activeCitations, highlightedId, setHighlightedId } = useReferenceStore();

  if (!isOpen) return null;

  return (
    <aside className={`flex flex-col bg-white border-l ${className}`}>
      {/* 头部 */}
      <div className="flex items-center justify-between p-4 border-b">
        <h2 className="text-lg font-semibold text-gray-900">引用来源</h2>
        <button
          onClick={onClose}
          className="p-1 rounded-lg hover:bg-gray-100 transition-colors"
        >
          <X className="w-5 h-5 text-gray-500" />
        </button>
      </div>

      {/* 引用列表 */}
      <div className="flex-1 overflow-y-auto p-4 space-y-4">
        {activeCitations.length === 0 ? (
          <div className="text-center text-gray-500 py-8">
            <p>暂无引用</p>
            <p className="text-sm mt-1">点击消息中的引用标记查看来源</p>
          </div>
        ) : (
          activeCitations.map((citation) => (
            <ReferenceCard
              key={citation.id}
              citation={citation}
              isHighlighted={citation.id === highlightedId}
              onClick={() => setHighlightedId(citation.id)}
            />
          ))
        )}
      </div>
    </aside>
  );
};

interface ReferenceCardProps {
  citation: Citation;
  isHighlighted: boolean;
  onClick: () => void;
}

const ReferenceCard: React.FC<ReferenceCardProps> = ({
  citation,
  isHighlighted,
  onClick,
}) => {
  const icons = {
    web: Globe,
    mcp: Database,
    memory: Brain,
    document: FileText,
  };

  const Icon = icons[citation.source_type] || Globe;

  return (
    <div
      onClick={onClick}
      className={`p-4 rounded-lg border cursor-pointer transition-all ${
        isHighlighted
          ? "border-primary-500 bg-primary-50 ring-2 ring-primary-200"
          : "border-gray-200 hover:border-gray-300 hover:bg-gray-50"
      }`}
    >
      {/* 头部 */}
      <div className="flex items-start justify-between gap-2">
        <div className="flex items-center gap-2">
          <span className="flex items-center justify-center w-6 h-6 rounded-full bg-primary-100 text-primary-700 text-sm font-medium">
            {citation.index}
          </span>
          <Icon className="w-4 h-4 text-gray-400" />
          <span className="text-xs font-medium text-gray-500 uppercase">
            {citation.source_type}
          </span>
        </div>
        {citation.url && (
          <a
            href={citation.url}
            target="_blank"
            rel="noopener noreferrer"
            className="p-1 rounded hover:bg-gray-200 transition-colors"
            onClick={(e) => e.stopPropagation()}
          >
            <ExternalLink className="w-4 h-4 text-gray-400" />
          </a>
        )}
      </div>

      {/* 标题 */}
      <h3 className="mt-2 font-medium text-gray-900 line-clamp-2">
        {citation.title}
      </h3>

      {/* URL */}
      {citation.url && (
        <p className="mt-1 text-xs text-gray-500 truncate">
          {citation.url}
        </p>
      )}

      {/* 摘要 */}
      <p className="mt-2 text-sm text-gray-600 line-clamp-3">
        {citation.snippet}
      </p>
    </div>
  );
};
```

## 2. 引用状态管理

### `frontend/src/stores/referenceStore.ts`

```typescript
import { create } from "zustand";
import type { Citation } from "@/types/session";

interface ReferenceState {
  // 当前激活的引用列表
  activeCitations: Citation[];
  // 高亮的引用 ID
  highlightedId: string | null;
  // 面板是否打开
  isPanelOpen: boolean;

  // 方法
  setActiveCitations: (citations: Citation[]) => void;
  addCitation: (citation: Citation) => void;
  setHighlightedId: (id: string | null) => void;
  setPanelOpen: (open: boolean) => void;
  highlightAndScroll: (id: string) => void;
  clearCitations: () => void;
}

export const useReferenceStore = create<ReferenceState>((set) => ({
  activeCitations: [],
  highlightedId: null,
  isPanelOpen: false,

  setActiveCitations: (citations) =>
    set({ activeCitations: citations }),

  addCitation: (citation) =>
    set((state) => ({
      activeCitations: [...state.activeCitations, citation],
    })),

  setHighlightedId: (id) =>
    set({ highlightedId: id }),

  setPanelOpen: (open) =>
    set({ isPanelOpen: open }),

  highlightAndScroll: (id) =>
    set({
      highlightedId: id,
      isPanelOpen: true,
    }),

  clearCitations: () =>
    set({ activeCitations: [], highlightedId: null }),
}));
```

```tsx
import { useEffect, useRef } from "react";
import { useChatStore } from "@/stores/chatStore";
import { MessageBubble } from "./MessageBubble";
import { InputBar } from "./InputBar";
import { ResearchProgress } from "./ResearchProgress";

export const ChatWindow: React.FC = () => {
  const { messages, isLoading, currentSession } = useChatStore();
  const messagesEndRef = useRef<HTMLDivElement>(null);

  // 滚动到底部
  const scrollToBottom = () => {
    messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
  };

  useEffect(() => {
    scrollToBottom();
  }, [messages]);

  return (
    <div className="flex flex-col h-full">
      {/* 标题栏 */}
      <div className="flex items-center justify-between px-6 py-4 border-b bg-white">
        <div>
          <h2 className="text-lg font-semibold text-gray-900">
            {currentSession?.title || "新对话"}
          </h2>
          {currentSession?.mode === "research" && (
            <span className="text-xs text-amber-600 font-medium">
              深度研究模式
            </span>
          )}
        </div>
      </div>

      {/* 研究进度（仅研究模式） */}
      {currentSession?.mode === "research" && <ResearchProgress />}

      {/* 消息列表 */}
      <div className="flex-1 overflow-y-auto px-6 py-4 space-y-6">
        {messages.length === 0 ? (
          <div className="text-center text-gray-500 py-12">
            <p className="text-lg font-medium">开始新的对话</p>
            <p className="text-sm mt-2">输入问题或切换到研究模式进行深度探索</p>
          </div>
        ) : (
          messages.map((message) => (
            <MessageBubble key={message.id} message={message} />
          ))
        )}
        {isLoading && (
          <div className="flex items-center gap-2 text-gray-500">
            <div className="w-2 h-2 bg-gray-400 rounded-full animate-bounce" />
            <div className="w-2 h-2 bg-gray-400 rounded-full animate-bounce delay-100" />
            <div className="w-2 h-2 bg-gray-400 rounded-full animate-bounce delay-200" />
          </div>
        )}
        <div ref={messagesEndRef} />
      </div>

      {/* 输入栏 */}
      <InputBar />
    </div>
  );
};
```

### `frontend/src/components/chat/MessageBubble.tsx`

```tsx
import ReactMarkdown from "react-markdown";
import remarkGfm from "remark-gfm";
import { User, Bot } from "lucide-react";
import { CitationLink } from "./CitationLink";
import { useReferenceStore } from "@/stores/referenceStore";
import type { Message } from "@/types/session";

interface MessageBubbleProps {
  message: Message;
}

export const MessageBubble: React.FC<MessageBubbleProps> = ({ message }) => {
  const isUser = message.role === "user";
  const { setActiveCitations, setPanelOpen, setHighlightedId } = useReferenceStore();

  // 处理引用点击
  const handleCitationClick = (citationId: string) => {
    if (message.citations) {
      setActiveCitations(message.citations);
      setPanelOpen(true);
      setHighlightedId(citationId);
    }
  };

  // 渲染带引用标记的内容
  const renderContent = (content: string) => {
    // 将 [n] 替换为 CitationLink 组件
    const parts = content.split(/(\[\d+\])/g);
    return parts.map((part, index) => {
      const match = part.match(/\[(\d+)\]/);
      if (match) {
        const citationId = match[1];
        return (
          <CitationLink
            key={index}
            index={parseInt(citationId)}
            onClick={() => handleCitationClick(citationId)}
          />
        );
      }
      return part;
    });
  };

  return (
    <div className={`flex gap-4 ${isUser ? "flex-row-reverse" : ""}`}>
      {/* 头像 */}
      <div
        className={`w-8 h-8 rounded-full flex items-center justify-center flex-shrink-0 ${
          isUser ? "bg-primary-100" : "bg-gray-100"
        }`}
      >
        {isUser ? (
          <User className="w-5 h-5 text-primary-600" />
        ) : (
          <Bot className="w-5 h-5 text-gray-600" />
        )}
      </div>

      {/* 内容 */}
      <div
        className={`max-w-[80%] rounded-2xl px-4 py-3 ${
          isUser
            ? "bg-primary-600 text-white"
            : "bg-white border border-gray-200"
        }`}
      >
        <div className={isUser ? "" : "prose prose-sm max-w-none"}>
          {isUser ? (
            <p className="text-white">{message.content}</p>
          ) : (
            <ReactMarkdown remarkPlugins={[remarkGfm]}>
              {message.content}
            </ReactMarkdown>
          )}
        </div>

        {/* 引用标记（仅助手消息） */}
        {!isUser && message.citations && message.citations.length > 0 && (
          <div className="flex flex-wrap gap-2 mt-3 pt-3 border-t border-gray-100">
            {message.citations.map((citation) => (
              <CitationLink
                key={citation.id}
                index={citation.index}
                onClick={() => handleCitationClick(citation.id)}
              />
            ))}
          </div>
        )}
      </div>
    </div>
  );
};
```

```tsx
import { BrowserRouter, Routes, Route, Navigate } from "react-router-dom";
import { AppLayout } from "./components/layout/AppLayout";
import { ChatWindow } from "./components/chat/ChatWindow";
import { ToolsConfig } from "./components/config/ToolsConfig";
import { SkillsConfig } from "./components/config/SkillsConfig";
import { MCPConfig } from "./components/config/MCPConfig";
import { LoginPage } from "./components/auth/LoginPage";
import { useAuthStore } from "./stores/authStore";

function App() {
  const { isAuthenticated } = useAuthStore();

  if (!isAuthenticated) {
    return <LoginPage />;
  }

  return (
    <BrowserRouter>
      <AppLayout>
        <Routes>
          <Route path="/chat" element={<ChatWindow />} />
          <Route path="/chat/:sessionId" element={<ChatWindow />} />
          <Route path="/config/tools" element={<ToolsConfig />} />
          <Route path="/config/skills" element={<SkillsConfig />} />
          <Route path="/config/mcp" element={<MCPConfig />} />
          <Route path="/" element={<Navigate to="/chat" replace />} />
        </Routes>
      </AppLayout>
    </BrowserRouter>
  );
}

export default App;
```
