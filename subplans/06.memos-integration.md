### `backend/app/services/memory_service.py`

```python
"""MemOS 记忆服务封装."""

from typing import Any

from app.config import settings


class MemoryService:
    """记忆服务，封装 MemOS SDK."""

    def __init__(self):
        self._pref_memory = None
        self._tree_memory = None
        self._feedback = None
        self._initialized = False

    async def initialize(self):
        """初始化 MemOS 客户端."""
        if self._initialized:
            return

        try:
            from memos.memories.textual.preference import PreferenceTextMemory
            from memos.memories.textual.tree import TreeTextMemory
            from memos.mem_feedback.simple_feedback import SimpleMemFeedback

            # PreferenceTextMemory 配置
            pref_config = {
                "vector_db": {
                    "backend": "milvus",
                    "host": settings.MILVUS_HOST,
                    "port": settings.MILVUS_PORT,
                    "collection_name": settings.MEMOS_PREF_COLLECTION,
                },
                "embedding": {
                    "model": settings.MEMOS_EMBEDDING_MODEL,
                    "api_url": settings.MEMOS_EMBEDDING_URL,
                },
            }

            # TreeTextMemory 配置
            tree_config = {
                "vector_db": {
                    "backend": "qdrant",
                    "host": settings.QDRANT_HOST,
                    "port": settings.QDRANT_PORT,
                    "grpc_port": settings.QDRANT_GRPC_PORT,
                    "collection_name": settings.MEMOS_TREE_COLLECTION,
                },
                "graph_db": {
                    "backend": "neo4j",
                    "uri": settings.NEO4J_URI,
                    "user": settings.NEO4J_USER,
                    "password": settings.NEO4J_PASSWORD,
                },
                "message_queue": {
                    "backend": "rabbitmq",
                    "url": settings.RABBITMQ_URL,
                },
                "embedding": {
                    "model": settings.MEMOS_EMBEDDING_MODEL,
                    "api_url": settings.MEMOS_EMBEDDING_URL,
                },
            }

            self._pref_memory = PreferenceTextMemory(pref_config)
            self._tree_memory = TreeTextMemory(tree_config)
            self._feedback = SimpleMemFeedback(
                preference_config=pref_config,
                tree_config=tree_config,
            )

            self._initialized = True

        except ImportError:
            # MemOS SDK 未安装，使用模拟实现
            print("Warning: MemOS SDK not found, using mock implementation")
            self._initialized = True

    # ========== 偏好记忆操作 ==========

    async def add_preference(
        self,
        messages: list[dict[str, str]],
        user_id: int,
        session_id: int,
        preference_type: str = "chat",
    ) -> list[dict[str, Any]]:
        """
        从对话中提取并存储用户偏好记忆.
        
        Args:
            messages: 对话消息列表
            user_id: 用户 ID
            session_id: 会话 ID
            preference_type: 偏好类型
        
        Returns:
            提取的记忆列表
        """
        if not self._initialized:
            await self.initialize()

        if self._pref_memory is None:
            return []

        try:
            memories = self._pref_memory.get_memory(
                messages=messages,
                type=preference_type,
                info={
                    "user_id": str(user_id),
                    "session_id": str(session_id),
                },
            )

            if memories:
                self._pref_memory.add(memories)

            return memories

        except Exception as e:
            print(f"Error adding preference: {e}")
            return []

    # ========== 树形记忆操作 ==========

    async def add_tree_memory(
        self,
        content: str,
        user_id: int,
        metadata: dict[str, Any] | None = None,
    ) -> dict[str, Any] | None:
        """
        添加树形结构记忆（知识点、研究成果）.
        
        Args:
            content: 记忆内容
            user_id: 用户 ID
            metadata: 元数据
        
        Returns:
            创建的记忆节点信息
        """
        if not self._initialized:
            await self.initialize()

        if self._tree_memory is None:
            return None

        try:
            user_name = f"user_{user_id}"
            memory_data = {
                "content": content,
                "metadata": metadata or {},
            }

            result = self._tree_memory.add([memory_data], user_name=user_name)
            return result[0] if result else None

        except Exception as e:
            print(f"Error adding tree memory: {e}")
            return None

    # ========== 记忆检索 ==========

    async def search(
        self,
        query: str,
        user_id: int | None = None,
        top_k: int = 10,
        search_type: str = "hybrid",  # "preference" | "tree" | "hybrid"
    ) -> list[dict[str, Any]]:
        """
        联合检索偏好记忆和树形记忆.
        
        Args:
            query: 查询文本
            user_id: 用户 ID（可选，用于过滤）
            top_k: 返回结果数量
            search_type: 搜索类型
        
        Returns:
            记忆列表
        """
        if not self._initialized:
            await self.initialize()

        results = []

        # 检索偏好记忆
        if search_type in ("preference", "hybrid") and self._pref_memory:
            try:
                pref_results = self._pref_memory.search(
                    query=query,
                    top_k=top_k // 2 if search_type == "hybrid" else top_k,
                    filter_dict={"user_id": str(user_id)} if user_id else None,
                )
                for r in pref_results:
                    r["source_type"] = "preference"
                results.extend(pref_results)
            except Exception as e:
                print(f"Error searching preference: {e}")

        # 检索树形记忆
        if search_type in ("tree", "hybrid") and self._tree_memory:
            try:
                user_name = f"user_{user_id}" if user_id else None
                tree_results = self._tree_memory.search(
                    query=query,
                    top_k=top_k // 2 if search_type == "hybrid" else top_k,
                    mode="hybrid",  # 向量 + 图搜索
                    user_name=user_name,
                )
                for r in tree_results:
                    r["source_type"] = "tree"
                results.extend(tree_results)
            except Exception as e:
                print(f"Error searching tree memory: {e}")

        # 按相关性排序
        results.sort(key=lambda x: x.get("score", 0), reverse=True)
        return results[:top_k]

    # ========== 记忆反馈 ==========

    async def process_feedback(
        self,
        user_id: int,
        session_id: int,
        chat_history: list[dict[str, str]],
        feedback_content: str,
    ) -> dict[str, Any]:
        """
        处理用户对记忆的反馈修正.
        
        Args:
            user_id: 用户 ID
            session_id: 会话 ID
            chat_history: 对话历史
            feedback_content: 反馈内容
        
        Returns:
            反馈处理结果
        """
        if not self._initialized:
            await self.initialize()

        if self._feedback is None:
            return {"status": "skipped", "reason": "feedback not available"}

        try:
            result = self._feedback.process_feedback(
                user_id=str(user_id),
                user_name=f"user_{user_id}",
                chat_history=chat_history,
                feedback_content=feedback_content,
            )
            return {"status": "success", "result": result}

        except Exception as e:
            print(f"Error processing feedback: {e}")
            return {"status": "error", "message": str(e)}


# 全局记忆服务实例
memory_service = MemoryService()
```

## 3. 记忆相关 API

### `backend/app/api/v1/memory.py`

```python
"""记忆管理 API 端点."""

from typing import Any

from fastapi import APIRouter, Query

from app.api.deps import CurrentUserDep
from app.services.memory_service import memory_service

router = APIRouter(prefix="/memory", tags=["memory"])


@router.get("/search")
async def search_memory(
    current_user: CurrentUserDep,
    query: str,
    top_k: int = Query(10, ge=1, le=50),
    search_type: str = Query("hybrid", regex="^(preference|tree|hybrid)$"),
):
    """
    搜索用户记忆.
    
    - preference: 只搜索偏好记忆
    - tree: 只搜索树形记忆
    - hybrid: 同时搜索两种记忆并合并排序
    """
    results = await memory_service.search(
        query=query,
        user_id=current_user.id,
        top_k=top_k,
        search_type=search_type,
    )

    return {
        "query": query,
        "total": len(results),
        "results": results,
    }


@router.post("/feedback")
async def memory_feedback(
    current_user: CurrentUserDep,
    session_id: int,
    feedback: str,
    chat_history: list[dict[str, str]],
):
    """
    提交记忆反馈.
    
    用户可以对记忆中的信息进行修正或补充.
    """
    result = await memory_service.process_feedback(
        user_id=current_user.id,
        session_id=session_id,
        chat_history=chat_history,
        feedback_content=feedback,
    )

    return result


@router.post("/tree")
async def add_tree_memory(
    current_user: CurrentUserDep,
    content: str,
    metadata: dict[str, Any] | None = None,
):
    """
    手动添加树形记忆.
    
    用户可以直接添加知识点到记忆树中.
    """
    result = await memory_service.add_tree_memory(
        content=content,
        user_id=current_user.id,
        metadata=metadata,
    )

    return {
        "status": "success" if result else "failed",
        "memory": result,
    }
```

## 5. 消息保存时提取记忆

### `backend/app/services/chat_service.py`

```python
"""聊天服务，处理消息保存和记忆提取."""

from app.services.memory_service import memory_service


class ChatService:
    """聊天服务."""

    @staticmethod
    async def save_message_and_extract_memory(
        session_id: int,
        user_id: int,
        messages: list[dict[str, str]],
    ) -> None:
        """
        保存消息后提取偏好记忆.
        
        在对话结束后异步调用此函数.
        """
        # 异步提取偏好记忆
        await memory_service.add_preference(
            messages=messages,
            user_id=user_id,
            session_id=session_id,
            preference_type="chat",
        )

    @staticmethod
    async def save_research_memory(
        user_id: int,
        query: str,
        report: str,
    ) -> None:
        """
        保存研究成果到树形记忆.
        """
        content = f"Research Query: {query}\n\nReport:\n{report[:2000]}"
        await memory_service.add_tree_memory(
            content=content,
            user_id=user_id,
            metadata={
                "type": "research",
                "query": query,
            },
        )
```

```python
"""ARQ Worker: 异步记忆处理任务."""

from arq import create_pool
from arq.connections import RedisSettings

from app.config import settings

arq_redis_settings = RedisSettings.from_dsn(settings.REDIS_URL)
from app.services.chat_service import ChatService


async def extract_preference_task(
    ctx: dict,
    session_id: int,
    user_id: int,
    messages: list[dict[str, str]],
) -> dict:
    """
    异步提取偏好记忆任务.

        user_id=user_id,
        messages=messages,

    ctx: dict,
    user_id: int,
    query: str,

        user_id=user_id,
        query=query,
        report=report,
    )
    return {"status": "completed", "user_id": user_id}


class MemoryWorkerSettings:

    redis_settings = RedisSettings.from_dsn(settings.REDIS_URL)
```

```typescript
  AddTreeMemoryRequest,
  MemoryFeedbackRequest,
```
